SELECT IFNULL(USER_NO, "NULL")        AS USER_NO    
       ,IFNULL(USER_ID, "NULL")        AS USER_ID
       ,IFNULL(USER_PWD, "NULL")        AS USER_PWD
       ,IFNULL(USER_TYPE, "NULL")        AS USER_TYPE
       ,IFNULL(USER_NAME, "NULL")        AS USER_NAME
       ,IFNULL(USER_EMAIL, "NULL")        AS USER_EMAIL
       ,IFNULL(USER_BIRTHDAY, "NULL")        AS USER_BIRTHDAY
       ,IFNULL(USER_COUNTRY, "NULL")        AS USER_COUNTRY
       ,IFNULL(USER_STATE, "NULL")        AS USER_STATE
       ,IFNULL(USER_ADDRESS, "NULL")        AS USER_ADDRESS
       ,IFNULL(USER_TEL_NO, "NULL")        AS USER_TEL_NO
       ,IFNULL(USER_SEX, "NULL")        AS USER_SEX
       ,IFNULL(USER_RGST_DT, "NULL")        AS USER_RGST_DT
       ,IFNULL(PUSH_TOKEN, "NULL")        AS PUSH_TOKEN
       ,IFNULL(LOGIN_DT, "NULL")        AS LOGIN_DT
       ,IFNULL(USER_DELETE_YN, "NULL")        AS USER_DELETE_YN
       ,IFNULL(USER_DELETE_DT, "NULL")        AS USER_DELETE_DT
       ,IFNULL(LOGIN_CK, "NULL")        AS LOGIN_CK
       ,IFNULL(COUNTRY_NM, "NULL")        AS COUNTRY_NM
       ,IFNULL(STATE_NM, "NULL")        AS STATE_NM
       ,IFNULL(SIDO_NM, "NULL")        AS SIDO_NM
       ,IFNULL(SIGUNGU_NM, "NULL")        AS SIGUNGU_NM
       ,IFNULL(EUPMYEONDONG_NM, "NULL")        AS EUPMYEONDONG_NM
       ,IFNULL(ADDR_DETAIL, "NULL")        AS ADDR_DETAIL
       ,IFNULL(ORGAN_CD, "NULL")        AS ORGAN_CD
       ,IFNULL(AUTH_STATUS, "NULL")        AS AUTH_STATUS
       ,IFNULL(AUTH_KEY, "NULL")        AS AUTH_KEY
       ,IFNULL(AUTH_EMAIL, "NULL")        AS AUTH_EMAIL
       ,IFNULL(TEETH_TYPE, "NULL")        AS TEETH_TYPE
FROM ST_USER
